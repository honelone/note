// 请求函数
const getList = () => {
  return new Promise((resolve, reject) => {
    //步骤一:创建异步对象
    var ajax = new XMLHttpRequest();
    //步骤二:设置请求的url参数,参数一是请求的类型,参数二是请求的url,可以带参数
    ajax.open('get', 'http://127.0.0.1:8000');
    //步骤三:发送请求
    ajax.send();
    //步骤四:注册事件 onreadystatechange 状态改变就会调用
    ajax.onreadystatechange = function () {
      if (ajax.readyState == 4 && ajax.status == 200) {
        //步骤五 如果能够进到这个判断 说明 数据 完美的回来了,并且请求的页面是存在的
        resolve(JSON.parse(ajax.responseText));
      }
    };
  });
};

// 获取container对象
const container = document.getElementById('container');

// 1. 直接渲染
// -- 非常耗费时间，且页面非常卡顿
const renderList1 = async () => {
  console.time('列表时间');
  const list = await getList();
  list.forEach((item) => {
    const div = document.createElement('div');
    div.className = 'sunshine';
    div.innerHTML = `<img src="${item.src}" /><span>${item.text}</span>`;
    container.appendChild(div);
  });
  console.timeEnd('列表时间');
};
// renderList1()

// 2. 分页渲染 setTimeout
// -- 将所有数据分为 page 页，然后每页渲染 total / page 条数据
// -- 然后再利用 setTimeout 每次渲染一页数据
// -- 相比直接渲染来说，缩短了首页渲染的时间
const renderList2 = async () => {
  console.time('列表时间');
  const list = await getList();
  console.log(list);
  const total = list.length;
  const page = 0;
  const limit = 200;
  const totalPage = Math.ceil(total / limit);

  const render = (page) => {
    // 大于页数则返回
    if (page >= totalPage) return;
    // 通过 setTimeout 事件机制来达到每次循环渲染一页数据
    setTimeout(() => {
      // 每执行一次，数据会从上次的结束位置开始
      for (let i = page * limit; i < page * limit + limit; i++) {
        const item = list[i];
        const div = document.createElement('div');
        div.className = 'sunshine';
        div.innerHTML = `<img src="${item.src}" /><span>${item.text}</span>`;
        container.appendChild(div);
      }
      render(page + 1);
    }, 0);
  };
  render(page);
  console.timeEnd('列表时间');
};
// renderList2()

// 3. 分页渲染 requestAnimationFrame
// -- 相比 setTimeout 来说，减少了 重排的次数，提高了性能
const renderList3 = async () => {
  console.time('列表时间');
  const list = await getList();
  console.log(list);
  const total = list.length;
  const page = 0;
  const limit = 200;
  const totalPage = Math.ceil(total / limit);

  const render = (page) => {
    if (page >= totalPage) return;
    // 使用requestAnimationFrame代替setTimeout
    requestAnimationFrame(() => {
      for (let i = page * limit; i < page * limit + limit; i++) {
        const item = list[i];
        const div = document.createElement('div');
        div.className = 'sunshine';
        div.innerHTML = `<img src="${item.src}" /><span>${item.text}</span>`;
        container.appendChild(div);
      }
      render(page + 1);
    });
  };
  render(page);
  console.timeEnd('列表时间');
};
// renderList3()

// 4. 分页渲染 文档碎片+requestAnimationFrame
// -- 上面都是每循环一次就创建一个 div 标签，并 appendChild 一次，这样会损耗性能
// -- 通过文档碎片，可以先将一页的 div 标签先放进文档碎片中，然后再一次性 appendChild 到 container 中，以减少 appendChild 的次数
// -- 页面只会渲染文档碎片包裹的元素，并不会渲染文档碎片
const renderList4 = async () => {
  console.time('列表时间');
  const list = await getList();
  console.log(list);
  const total = list.length;
  const page = 0;
  const limit = 200;
  const totalPage = Math.ceil(total / limit);

  const render = (page) => {
    if (page >= totalPage) return;
    requestAnimationFrame(() => {
      // 创建一个文档碎片 fragment
      const fragment = document.createDocumentFragment();
      for (let i = page * limit; i < page * limit + limit; i++) {
        const item = list[i];
        const div = document.createElement('div');
        div.className = 'sunshine';
        div.innerHTML = `<img src="${item.src}" /><span>${item.text}</span>`;
        // 这里先将我们创建的 div 标签 塞进文档碎片
        fragment.appendChild(div);
      }
      // 最后再一次性将文档碎片 appendChild 到 container 中
      container.appendChild(fragment);
      render(page + 1);
    });
  };
  render(page);
  console.timeEnd('列表时间');
};
// renderList4()
