更新渲染 ​

我们都知道Vue的数据响应式是通过数据劫持和观察订阅模式来实现的

而这是实现数据更新的同时页面也更新的关键

即由监听器Observe发出通知给订阅器Dep，订阅器再发出通知给订阅者Watcher

01. 订阅数据变化通知 ​

  • 在定义响应式数据时，也就是在数据劫持时，我们需要给每个进行响应式处理的据添加一个订阅器

    • 这里会通过Dep类来实例化一个订阅器，然后还会用到订阅器里面的一些方法和属性

    • Dep.target：这个属性表示当前实例所指向的订阅者，也即Watcher

    • dep.depend()：这个方法用于收集依赖，也即将当前的订阅器放入订阅者中

    • dep.notify()：这个方法用于发出通知给订阅者，让订阅者进行更新

      更具体的请继续往下看，这里只是简单介绍下它们的用途

    js
    // ./observe/index.js

function defineReactive(data, key, value) {
    observe(value)
    // 每对一个数据进行响应式处理时，都给它创建一个订阅器+
    // -- 它表示的是：这个数据之后的响应式变化都会被这个订阅器收集
    let dep = new Dep()

    Object.defineProperty(data, key, {
        get() {
            // 在获取值的时候，我们会将其收集到依赖中
            // -- 表示的是，这个数据已经在使用中了，
            // -- 后续我们会订阅这个数据的变化
            if(Dep.target) {
                // 调用订阅器的方法 depend 来收集这个依赖
                // -- 而这个依赖会被收集到相应的 watcher 监听者中
                dep.depend()
            }
            reutn value
        },
        set(newValue) {
            if(newValue === value) return
            // 这里补充一下，如果新的值是一个对象，那么要继续去观测
            observe(newValue)
            // 设置新值
            value = newValue
            // 在改变了值后，就要发出通知给订阅器
            // -- 调用订阅器的 notify 方法
            // -- 再由订阅器的 notify 去通知 watcher 订阅者去更新
            dep.notify()
        }
    })
}

    let childOb = observe(value)

        },

            observe(newValue)
            value = newValue
            dep.notify()
        }
    })
}

  • 对数组进行依赖收集

    js
    // 递归收集数组依赖
function dependArray(value) {
  const length = value.length;
  let e;
  for (let i = 0; i < length; i++) {
    e = value[i];
    // e.__ob__代表e已经被响应式观测了,但是没有收集依赖 所以把他们收集到自己的Observer实例的dep里面
    e && e.__ob__ && e.__ob__.dep.depend();

    // 递归收集依赖
    if (Array.isArray(e)) {
      dependArray(e);
    }
  }
}

02. 订阅器Dep ​

  • 好了，上面介绍了那么多订阅器，那么现在我们来看看订阅器究竟是个什么东西

    • 订阅器主要用于：订阅数据变化，收集订阅者（观察者）watcher
      • 在数据变化时就会发出通知dep.notify()给订阅者（观察者）
      • 订阅者（观察者）收到这个通知后就会去重新调用_update()方法以更新页面
    • 至于如何更新的，这个会在后面创建订阅者（观察者）Watcher的时候讲到

  • 现在，让我们把目光聚焦到订阅器，

    • 订阅器主要有三个方法

      • depend：收集依赖
      • notify：发出通知
      • addSub：新增订阅者

      具体的介绍会在代码里面注释

    • 而在订阅器里还会用到订阅者（观察者）的一些方法

      • addDep：用于把当前Dep实例放入订阅者Watcher中
      • update：用于在收到通知时，执行更新方法
    js
    // ./observer/dep.js

// 自增变量，用于标识每一个订阅器
let id = 0;
export default class Dep {
  constructor() {
    this.id = id++;
    this.subs = []; // subs 即用于存放 watcher 的数组
  }
  // 1. 这个方法用于收集依赖
  // -- 那么什么是收集依赖呢 QAQ
  // -- 简单来说，就是页面模板上用到了（依赖了）data中的哪些数据
  // -- 这些数据就会被收集到 watcher 中
  // -- 而如何知道模板中需要哪些数据呢？
  // -- 答案就是在数据劫持时，在 getter 中进行判断
  // -- 当模板中用到数据时，就会触发 getter 函数
  // -- 从而收集相关依赖
  // -- 这里具体的会在后面 Watcher 中讲到
  depend() {
    // -- 而如何进行判断的关键就在于这个 Dep.target 了
    // -- 这个属性是 Dep 的全局变量
    // -- 这个变量指向当前的 watcher
    // -- 具体的下面有讲到
    if (Dep.target) {
      // 这里表示如果当前 watcher 存在
      // -- 那么就调用 Watcher 上的方法收集依赖
      // -- 为什么是收集订阅器 Dep?
      // -- 因为最终由是 Dep 来发出更新通知的
      // -- 收集了这个订阅器，就表示会接受这个订阅器发出的通知
      Dep.target.addDep(this);
    }
  }
  // 2. 这个方法用于发出通知
  // -- 具体的就是在调用这个方法时，遍历当前 subs 数组
  // -- 然后对每一个订阅者都调用其 upodate 方法
  // -- 以此来达到更新的目的
  notify() {
    this.subs.forEach((watcher) => watcher.update());
  }
  // 3. 收集订阅者
  // -- 这个方法在上面数据劫持时没有使用
  // -- 这是因为这个方法是在 Watcher 里面才能用到的
  // -- subs 用于存放订阅者，而订阅者当然只有在 Watcher 中才有啊
  // -- 至于这个方法具体在哪里使用，后续会讲到
  addSub(watcher) {
    this.subs.push(watcher);
  }
}

// 默认的 Dep.target 指向 null
// -- Dep.target是一个全局的 Watcher 指向
Dep.target = null;

// Dep.target 主要通过下面两个方法进行更新
// -- 这两个方法主要会在 Watcher 中使用到
// -- 毕竟，这个是用来指向 Watcher 的 @_@

// 先创建一个数组用来存储所有的 watcher
const targetStack = [];
export function pushTarget(watcher) {
  // 先存入数组中
  targetStack.push(watcher);
  // 然后让 Dep.target 指向当前 watcher
  Dep.target = watcher;
}
export function popTarget(watcher) {
  // 将数组的最后一个 watcher 移除
  targetStack.pop();
  // 然后让 Dep.target 指向移除后的数组最后一个 Watcher
  Dep.target = targetStack[targetStack.length - 1];
}

03. 观察者Watcher ​

  • 接下来，我们一起来看看观察者模式

    • 首先要有一个大类Watcher，对每一个数据都会生成一个观察者
    js
    // observer/watcher.js
import { pushTarget, popTarget } from './dep.js';
// 这个 id 用于标识每一个 Watcher 实例，每 new 一个 Watcher ,id 都会自增
let id = 0;
export default class Watcher {
  constructor(vm, exprOrFn, cb, options) {
    this.vm = vm; // 即当前 Vue 实例
    this.exprOrFn = exprOrFn; //
    this.cb = cb; // 回调函数
    this.options = options; // 额外的参数, true 表示渲染 watcher
    this.id = id++; // 唯一标识
    this.deps = []; // 用于存放订阅器 dep
    this.depsId = new Set(); // 用于去重 dep
    // 如果表达式是一个函数
    if (typeof exprOrFn === 'Function') {
      this.getter = exprOrFn;
    }

    this.get(); // 实例化时会调用这个方法
  }
  get() {
    // 调用 getter 之前，需要先把当前的 watcher 推到 Dep.target上
    pushTarget(this);
    // 调用这个方法以更新渲染视图
    this.getter();
    // 调用 getter 之后，需要把当前 watcher 移除
    popTarget(this);
  }
  // 收集依赖，将订阅器收集起来
  addDep(dep) {
    let id = dep.id;
    if (!this.depsId.has(id)) {
      this.depsId.add(id);
      this.deps.push(dep);
      // 然后，将当前 watcher 加入到 dep 的 subs 中
      // -- 订阅器中的 addSub 用于这里
      dep.addSub(this);
    }
  }
  // 更新
  update() {
    this.get();
  }
}

export function mountComponent(vm, el) {
  vm.$el = el;
  // updateComponent 即返回的真实DOM

  let updateComponent = () => {
    vm._update(vm._render());

  // 创建一个观察者实例
  new Watcher(vm, updateComponent, null, true);
