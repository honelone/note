02. 路由组件映射 ​

  • <Routes>：包裹所有的<Route>在其中匹配一个路由
    • Router5.x使用的是<Switch>组件
  • <Route>：用于路径的匹配，所有的Route必须放在Routes里
    • path属性：用于设置匹配到的路径
    • element属性：设置匹配到路径后，渲染的组件
      • Router5.x使用的是component属性
    • exact属性：精准匹配，只有完全一致的路径才会渲染
      • Router6.x不再使用
jsx
class App extends PureComponent {
  render() {
    return (
      <Fragment>
        <Routes>
          <Route path="/home" element={<Home />} />
          <Route path="/about" element={<About />} />
        </Routes>
      </Fragment>
    );
  }
}

03. 路由组件跳转 ​

  • <Link>：最终会被渲染为<a>元素
  • <NavLink>：在<Link>的基础上增加了一些样式属性
    • 当选中时，会默认添加一个active类
    • 也可以通过style | className传入一个函数来自定义类
    • 这里仅作了解，很少用
  • 属性
    • to属性：用于设置跳转的目标路径
    • replace属性：路径替换
    • reloadDocument属性：跳转后是否重新加载页面
jsx
class App extends PureComponent {
  render() {
    return (
      <div>
        <Link to="/home">Home Page</Link>
        <Link to="/about">About Page</Link>
      </div>
    );
  }
}

04. 路由重定向 ​

  • 使用<Navigate>组件，用于路由的重定向

    • 当这个组件出现，就会跳转到组件上配置的to属性所对应的路径
    • Router5.x中使用的是<Redirect>

  • 多用于判断用户是否登录，跳转首页或登陆页

    jsx
    class App extends PureComponent {
  render() {
    return (
      <Fragment>
        <Routes>
          {/* 将 / 重定向到 /home */}
          <Route path="/" element={<Navigate to="/home" />} />
          <Route path="/home" element={<Home />} />
        </Routes>
      </Fragment>
    );
  }
}

  • Router5.x：

  • Router6.x：

    • 使用useNavigate，获取到navigate对象
    • 但只能在函数式组件中使用
      • 直接传入跳转路径
      • 传入数字
      • 传入对象
        • to属性：跳转路径

    jsx
    import { useNavigate } from 'react-router-dom'

// 函数式组件
export default function withRouter(WrapperComponent) {
    // 只能放在顶层
    const navigate = useNavigate()

    function navigateTo(path){
        navigate(path)
    }
    return props => {
        return (
            <div>
                <button onClick={ e => navigateTo('/home') }></button>
            </div>
        )
    }
}

// 高阶组件
import { useNavigate } from 'react-router-dom'
export default function withRouter(OriginComponent) {
    return ( props ) => {
        const navigate = useNavigate()
        const router = {
            navigate
        }

        return <OriginComponent {...props} router={router} />
    }
}

// 类组件 -- 需要通过高阶组件进行 强化

08. 路由传参 ​

  • 两种方式

    • 动态路由传参：在路由后接参数

      • 需要将路由匹配写成/home/:id的形式
      • 然后/home/123、/home/456都将匹配到同一个组件
      • 这个参数可以通过useParams获取到
      jsx
      // 需要 将 useParams 加入到 高阶组件中
import { useNavigate, useParams } from 'react-router-dom'
export default function withRouter(OriginComponent) {
    return props => {
        const navigate = useNavigate()
        const params = useParams()
        const router = {
            navigate,
            params
        }

        return <OriginComponent {...props} router={router} />
    }
}

import { useNavigate, useParams, useLocation, useSearchParams } from 'react-router-dom';

export default function withRouter(OriginComponent) {
  return (props) => {
    const navigate = useNavigate();
    const params = useParams();
    const location = useLocation();

    const [searchParams] = useSearchParams();
    const query = Object.fromEntries(searchParams);

    const router = {
      navigate,
      params,
      location,
      query,
    };

    return <OriginComponent {...props} router={router} />;
  };
}
