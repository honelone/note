03. 图片手动上传组件封装 ​

  • 基于Ant-Design-Vue

    vue
    <template>
  <div class="upload-img-container">
    <a-upload
      v-bind="{ ...$attrs }"
      action="#"
      accept=".jpg, .png, .jpeg"
      list-type="picture-card"
      :show-upload-list="false"
      :beforeUpload="beforeUpload"
      :customRequest="customRequest"
    >
      <!-- 上传 -->
      <div v-if="imgList.length < size">
        <a-icon type="plus" />
        <div class="ant-upload-text">上传</div>
      </div>
      <!-- 图片列表 -- 这里只有一张图片，所以放在组件里面，如果需要多个图片，则需要放在外面 -->
      <template v-else>
        <div class="list-item" v-for="(item, index) in imgList" :key="item.url">
          <img v-if="item.type == 'origin'" :src="item.url" alt="相关图片" />
          <img v-else :src="item.url" alt="相关图片" />
          <!-- 预览 -->
          <div class="item-hover">
            <a-icon style="margin-right: 10px" type="eye" @click.stop="openPreview(index)" />
            <a-icon type="delete" @click.stop="deleteRelateImg(index)" />
          </div>
        </div>
      </template>
    </a-upload>
    <!-- 预览弹窗 -->
    <a-modal :visible="previewVisible" :footer="null" @cancel="closePreview" :centered="true">
      <img alt="example" style="width: 100%" :src="previewImage" />
    </a-modal>
  </div>
</template>

<script>
  export default {
    name: 'UploadImg',
    props: {
      // 图片数量
      size: {
        type: Number,
        default: 1,
      },
      // 回显图片数据
      // -- 这里是字符串是因为这里只显示一张图片
      img: {
        type: String,
        default: '',
      },
    },
    watch: {
      // 处理回显
      img(newValue) {
        if (!newValue) return;
        this.imgList = [];
        this.imgList.push({
          type: 'upload',
          url: newValue,
        });
      },
    },
    data() {
      return {
        imgList: [], // 图片数组
        previewVisible: false, // 预览图片弹窗
        previewImage: '', // 预览图片
      };
    },
    methods: {
      // 上传前校验 -- 校验格式和大小
      beforeUpload(file) {
        const isJpgOrPng =
          file.type === 'image/jpeg' || file.type === 'image/jpg' || file.type === 'image/png';
        if (!isJpgOrPng) {
          this.$message.error('只能上传jpg/png格式的图片!');
        }
        const isLt2M = file.size / 1024 / 1024 < 2;
        if (!isLt2M) {
          this.$message.error('图片不得大于2MB!');
        }
        return isJpgOrPng && isLt2M;
      },
      // 自定义上传获取文件内容
      async customRequest(fileInfo) {
        const { file } = fileInfo;
        try {
          const url = await this.fileToBase64(file); // 获取base64地址
          // -- 通过 type 来标识是当前上传的，还是后台返回的
          // -- 这里暂只作上传的处理
          // -- file 暂无用
          // -- url 为图片 base64地址
          let ob = {
            type: 'upload',
            url,
            file,
          };
          // 推入当前图片数组
          this.imgList.push(ob);
          // 更新父组件
          this.$emit('update:img', ob.url);
        } catch (err) {
          this.$message.error(err.message);
        }
      },
      // 转换图片地址
      fileToBase64(file) {
        const reader = new FileReader();
        reader.readAsDataURL(file);
        return new Promise((resolve, reject) => {
          reader.onload = function (e) {
            if (this.result) {
              resolve(this.result);
            } else {
              reject('图片转换错误，请稍后重试');
            }
          };
        });
      },
      // 以下为图片预览相关事件 ======
      // 照片墙预览
      openPreview(index) {
        // 如果是服务器返回的，则拼接地址
        // 如果是自己上传的，则直接显示base64地址
        // -- 这个根据后端返回的图片路径判断
        if (this.imgList[index].type === 'upload') {
          this.previewImage = this.imgList[index].url;
        } else {
          // baseUrl 需要自定义
          // this.previewImage = this.baseUrl + this.imgList[index].url
        }

        this.previewVisible = true;
      },
      // 关闭预览图片遮罩层
      closePreview() {
        this.previewVisible = false;
      },
      // 删除上传图片
      deleteRelateImg(index) {
        this.$confirm({
          title: '确定删除该图片?',
          okType: 'danger',
          onOk: () => {
            // 删除并更新父组件
            this.imgList.splice(index, 1);
            this.$emit('update:img', '');
          },
        });
      },
    },
  };
</script>

05. JS 计算精度解决方法 ​

  • 进制转换 ：
    • js在做数字计算的时候，0.1 和 0.2 都会被转成二进制后无限循环
    • 但是js采用的 IEEE 754 二进制浮点运算，最大可以存储 53 位有效数字
    • 于是大于 53 位后面的会全部截掉，将导致精度丢失
  • 对阶运算 ：
    • 由于指数位数不相同，运算时需要对阶运算，阶小的尾数要根据阶差来右移（0舍1入）
    • 尾数位移时可能会发生数丢失的情况，影响精度
js
# math.js

/*
 * 思路：把数字转换成字符串，然后以小数点分割为两部分
 * 算出小数点部分的长度n，然后小数部分乘以 10 的 n 次方，放大倍数
 * 即，将两个浮点数放大为整数进行相加（相减），之后再缩小倍数
 * 对于乘法也是将其转换为整数进行处理
 */

// 加法
const add = (num1, num2) => {
	const num1Digits = (num1.toString().split('.')[1] || '').length
	const num2Digits = (num2.toString().split('.')[1] || '').length
	const baseNum = Math.pow(10, Math.max(num1Digits, num2Digits))
	return (times(num1, baseNum) + times(num2, baseNum)) / baseNum
}

// 减法
const subtract = (num1, num2) => {
	const num1Digits = (num1.toString().split('.')[1] || '').length
	const num2Digits = (num2.toString().split('.')[1] || '').length
	const baseNum = Math.pow(10, Math.max(num1Digits, num2Digits))
	return (times(num1, baseNum) - times(num2, baseNum)) / baseNum
}

// 乘法
const times = (num1, num2) => {
	const num1String = num1.toString()
	const num2String = num2.toString()
	const num1Digits = (num1String.split('.')[1] || '').length
	const num2Digits = (num2String.split('.')[1] || '').length
	const baseNum = Math.pow(10, num1Digits + num2Digits)
	return (Number(num1String.replace('.', '')) * Number(num2String.replace('.', ''))) / baseNum
}

// 除法
const divide = (num1, num2) => {
	const num1String = num1.toString()
	const num2String = num2.toString()
	const num1Digits = (num1String.split('.')[1] || '').length
	const num2Digits = (num2String.split('.')[1] || '').length
	const baseNum = Math.pow(10, num1Digits + num2Digits)
	let n1 = floatMultiply(num1, baseNum)
	let n2 = floatMultiply(num2, baseNum)
	return Number(n1) / Number(n2)
}

export const math = {
	add,
	subtract,
	times,
	divide,
}

（4）强化版 ​

  • 其它数据类型
    • 上面只考虑了对象和数组两种类型，还有许多其它的引用类型，所以要获取数据的具体类型，详见类型
  • 然后这些类型又可分为可继续遍历类型、不可继续遍历类型
js
const mapTag = '[object Map]';
const setTag = '[object Set]';
const arrayTag = '[object Array]';
const objectTag = '[object Object]';
const argsTag = '[object Arguments]';

const boolTag = '[object Boolean]';
const dateTag = '[object Date]';
const numberTag = '[object Number]';
const stringTag = '[object String]';
const symbolTag = '[object Symbol]';
const errorTag = '[object Error]';
const regexpTag = '[object RegExp]';
const funcTag = '[object Function]';

const deepTag = [mapTag, setTag, arrayTag, objectTag, argsTag];

function forEach(array, iteratee) {
  let index = -1;
  const length = array.length;
  while (++index < length) {
    iteratee(array[index], index);
  }
  return array;
}

function isObject(target) {
  const type = typeof target;
  return target !== null && (type === 'object' || type === 'function');
}

function getType(target) {
  return Object.prototype.toString.call(target);
}

function getInit(target) {
  const Ctor = target.constructor;
  return new Ctor();
}

function cloneSymbol(targe) {
  return Object(Symbol.prototype.valueOf.call(targe));
}

function cloneReg(targe) {
  const reFlags = /\w*$/;
  const result = new targe.constructor(targe.source, reFlags.exec(targe));
  result.lastIndex = targe.lastIndex;
  return result;
}

function cloneFunction(func) {
  const bodyReg = /(?<={)(.|\n)+(?=})/m;
  const paramReg = /(?<=\().+(?=\)\s+{)/;
  const funcString = func.toString();
  if (func.prototype) {
    const param = paramReg.exec(funcString);
    const body = bodyReg.exec(funcString);
    if (body) {
      if (param) {
        const paramArr = param[0].split(',');
        return new Function(...paramArr, body[0]);
      } else {
        return new Function(body[0]);
      }
    } else {
      return null;
    }
  } else {
    return eval(funcString);
  }
}

function cloneOtherType(targe, type) {
  const Ctor = targe.constructor;
  switch (type) {
    case boolTag:
    case numberTag:
    case stringTag:
    case errorTag:
    case dateTag:
      return new Ctor(targe);
    case regexpTag:
      return cloneReg(targe);
    case symbolTag:
      return cloneSymbol(targe);
    case funcTag:
      return cloneFunction(targe);
    default:
      return null;
  }
}

function clone(target, map = new WeakMap()) {
  // 克隆原始类型
  if (!isObject(target)) {
    return target;
  }

  // 初始化
  const type = getType(target);
  let cloneTarget;
  if (deepTag.includes(type)) {
    cloneTarget = getInit(target, type);
  } else {
    return cloneOtherType(target, type);
  }

  // 防止循环引用
  if (map.get(target)) {
    return map.get(target);
  }
  map.set(target, cloneTarget);

  // 克隆set
  if (type === setTag) {
    target.forEach((value) => {
      cloneTarget.add(clone(value, map));
    });
    return cloneTarget;
  }

  // 克隆map
  if (type === mapTag) {
    target.forEach((value, key) => {
      cloneTarget.set(key, clone(value, map));
    });
    return cloneTarget;
  }

  // 克隆对象和数组
  const keys = type === arrayTag ? undefined : Object.keys(target);
  forEach(keys || target, (value, key) => {
    if (keys) {
      key = value;
    }
    cloneTarget[key] = clone(target[key], map);
  });

  return cloneTarget;
}

08. 自定义打印方法 ​

  • 通过iframe来调用打印
js
/**
 * 打印
 * @param {string} contentId 组件id
 * @param {string} printCss 样式名
 */
export const goToPrint = function (contentId, printCss = null) {
  let w,
    doc,
    iframe = document.createElement('iframe'),
    f = document.body.appendChild(iframe);

  iframe.id = 'myIframe';
  iframe.setAttribute('style', 'position:absolute;width:0;height:0;top:-500px;left:-500px;');
  w = f.contentWindow || f.contentDocument;
  doc = f.contentDocument || f?.contentWindow?.document;

  let el = document.getElementById(contentId);

  // 打开
  doc.open();
  // 如果有样式传入，则 写入样式，注：可自定义样式位置，以 public 为基本路劲
  if (!!printCss) {
    let path = '';
    if (process.env.FLAG == 'dev') {
      path = '/css/';
    } else {
      path = '/mic/workflow/css/';
    }
    let str = '<LINK rel="stylesheet" type="text/css" href="' + path + printCss + '">';
    doc.write(str);
  }

  // 设置打印窗口属性
  doc.write('<style media="print">@page{size:auto A4 landscape;margin:5mm 8mm;}</style>');

  // 设置传入的元素内容
  doc.write('<div>' + el.innerHTML + '</div>');
  // 关闭
  doc.close();
  // 聚焦
  iframe.contentWindow.focus();
  // 加载
  iframe.onload = function () {
    try {
      setTimeout(function () {
        w.focus();
        try {
          if (!w.document.execCommand('print', false, null)) {
            w.print();
          }
        } catch (e) {
          w.print();
        }
        w.close();
      }, 100);
    } catch (err) {
      console.log('err', err);
    }
    setTimeout(function () {
      document.body.removeChild(iframe);
    }, 100);
  };
};
