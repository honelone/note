01. setup ​

（1）ref的用法 ​

typescript
import { ref, watch } from 'vue';

// ref -- 适用于简单类型的变量
const name = ref<string>('Composition');
const age = ref<number>(24);
const hasMoney = ref<boolean>(false);
// -- 使用需要带上 .value
age.value = 28;

// 数组一般也用 ref 来定义
const hobbies = ref<string[]>(['code', 'eat', 'sleep']);
// -- 需要注意的是：不要使用 数组方法 更新数组，需要直接赋值
hobbies.value = [...hobbies.value, 'note'];

// 对 ref 类型的数据的监听, 直接监听
watch(age, () => {
    console.log(age.value)
})

（2）reactive的用法 ​

typescript
import { reactive, watch } from 'vue';

// 一般从外部引入类型
interface IUserInfo {
    name: string;
    age: number;
}

// reactive -- 适用于引用类型的变量
const userInfo = reactive<IUserInfo>({
    name: 'Composition',
    age: 24,
})

// 对 reative的监听
// -- 整个对象的监听
watch(userInfo, () => {
    console.log(userInfo)
})
// -- 对某个属性的监听
watch(() => userInfo.age, () => {
    console.log(userInfo.age)
})

// 对于 reactive 的改变，不能直接使用 = 赋值，需要具体到某个属性
const setUserInfo1 = (newUserInfo: IUserInfo) => {
    userInfo.name = newUserInfo.name;
    userInfo.age = newUserInfo.age;
}
//-- 对于属性太多的情况，可以使用 Object.assign()
const setUserInfo2 = ((newUserInfo: IUserInfo) => {
    Object.assign(userInfo, newUserInfo);
}

（3）props、emits、computed的使用 ​

typescript
import { computed } from 'vue';

// 接收父级组件传进来的参数， 可以使用 withDefaults 来设置默认值
const props = defineProps<{
    name: string;
    age: number;
}>()
const propsWithDefaults = withDefaults(
    defineProps<{
        name: string;
        age: number;
    }>(), {
        name: 'Composition',
        age: 28,
    })

const emits = defineEmits<{
    (event: 'update:name', newVal): void;
    (event: 'update:age', newVal): void;
}>()

// 计算属性返回的是一个 ref 类型的数据
const currentName = computed({
    get() {
        return props.name;
    },
    set(newVal) {
        emits('update:name', newVal)
    }
})

const currentAge = computed({
    get() {
        return props.age;
    },
    set(newVal) {
        emits('update:age', newVal)
    }
})

（4）provide、inject的使用 ​

typescript
import { ref, reactive, provide, inject } from 'vue';

// 第二个参数为可选参数，默认值
const parentParams = inject('parentParams', ref<string>(''));
// 用于表示当前为 Function 类型的函数
const parentFunc = inject('parentFunc', Function, true);

const state = reactive({});
provide('state', state);

（5）ref和defineExpose ​

typescript
import { ref } from 'vue';
import Child from './Child.vue';

// ref 还可定义为组件实例
// -- 普通标签
const userRef = ref<HTMLElement | null>(null);
// -- 自定义组件
const childRef = ref<InstanceType<typeof Child> | null>(null);
// -- 不关心类型，则可以使用 ComponentPublicInstance
const otherRef = ref<ComponentPublicInstance | null>(null)

// 只有在子组件内部通过 defineExpose 导出去的变量才能在子组件的实例上访问到
const _id = ref('123');
defineExpose({
    _id,
})

02. setup + TS ​

（1）单文件模板 ​

  • 有时候会需要在一个文件中使用单文件组件模板，但又不想再写.vue文件了

    • 可以使用.tsx文件的写法，直接重写整个组件

    • 在<script setup>上添加属性lang="tsx"

      • 这样就可以直接定义函数式组件了
      vue
      <script lang="tsx" setup>
    const customComponent = () => {
        return (<div>Custom </div>>);
    };
</script>

02. 引入第三方包的需求 ​

  • 目前是有一个阿里云滑动块验证的需求，需要引入阿里云的第三方文件

    html
    <script src="https://g.alicdn.com/AWSC/AWSC/awsc.js"></script>
    1

  • 在Vue3中使用：

    • 一种是直接在index.html中引入
    • 一种是在需要的文件中动态创建<script>标签引入
    js
    const loadScript = (callback) => {
    const script = document.createElement('script');
    script.src = 'https://g.alicdn.com/AWSC/AWSC/awsc.js';
    document.getElementsByTagName('head')[0].appendChild(script);
    // 引入成功事件
    script.onload = () => {
        console.log('load success')
        callback()
    }
    // 引入失败事件
    script.onerror = () => {
        console.log('load error')
    }
    
}
