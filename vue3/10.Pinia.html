（2）定义 ​

js
// 定义
import { defineStore } from 'pinia'

export const mainStore = defineStore('main', {
  state: () => {
    return {
      name: '',
      age: '',
    }
  },
  getters: {
    getName() {
      return this.name
    },
    getAge: state => {
      return state.age
    },
  },
  actions: {
    changeName(newName) {
      this.name = newName
    },
    changeAge(newAge) {
      this.age = newAge
    },
  },
})

（5）修改 ​

js
// 修改
import { mainStore } from '../store/index'

const store = mainStore()

// 直接通过属性名修改
store.name = 'new name'

// 通过 &patch 的方法去修改，性能更好
// 对象形式
store.$patch({
  name: 'new name',
  age: 18,
})
// 函数形式
store.$patch(state => {
  state.name = 'new name'
  state.age = 18
})

// 通过 action 中的方法修改
store.changeName('new name')
store.changeAge(18)

（6）监听订阅 ​

js
// 监听订阅
import { mainStore } from '../store/index'

const store = mainStore()

// 通过 $subscribe 可以监听 state 变化
const stopSubscribe = store.$subscribe((mutation, state) => {
  // mutation -- 一个对象，包含 events, storeId, type
  // state -- mainStore 的实例
})

stopSubscribe() // 调用返回值可以停止监听

// 通过 $onAction 可以监听 action 触发
const stopOnAction = mainStore.$onAction(
  ({
    name, // action 函数的名称
    store, // store 实例，这里是 mainStore
    args, // action 函数参数数组
    after, // 钩子函数，在action函数执行完成返回或者resolves后执行
    onError, // 钩子函数，在action函数报错或者rejects后执行
  }) => {
    // do sth...
  }
)
stopOnAction() // 调用返回值可以停止监听

（7）插件

js
// pinia 插件
import { createPinia } from 'pinia'

const pinia = createPinia()

pinia.use(() => {
    env: 'env',
})

app.use(pinia)

// 所有其它 store 都可以访问到这个 env 属性

// ---------
// 参数
pinia.use(({ app, options, pinia, store }) => {
    // app -- 当前应用 Vue.createApp() 创建的 app
    // options -- defineStore 配置的数据
    // pinia -- 当前通过 createPinia() 创建的 pinia 实例
    // store -- 当前 store 实例
})

// 内部监听
pinia.use(({ store }) => {
    store.$suscribe(() => {})

    store.$onAction(() => {})
})
